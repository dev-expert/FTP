#include <iostream>

using namespace std;
 int myfunction(int arr[],int n)
    {
        int i,j;
    for(i=0;i<n;i++)
    {
        for(j=1;j<n;j++)
        {
           if(arr[i]<arr[j] && arr[j]<arr[j+1])
           {
               return arr[j];
               
           }
        }
    }
    }

int main()
{
    int n,i;
    cout<<"Enter number";
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    cout<<myfunction(arr,n);
}
