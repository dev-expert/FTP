#include<bits/stdc++.h>

using namespace std;



int main() {
    float positive=0,negative=0,zero=0;
    int n, i;
   
    cout<<"Enter number of elements: ";
    cin>>n;
     
    int arr[n];
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
	for(i=0;i<n;i++)
	{
	    if(arr[i]>0)
	    {
	        positive = positive+1;
	    }
	    else if(arr[i]<0){
	        negative=negative+1;
	    }
	    else if(arr[i]==0){
	        zero=zero+1;
	    }
	}
	positive=positive/n;
	negative=negative/n;
	zero=zero/n;
	cout<<"Postive Values Ratio: ";
    printf("%.6lf",positive);
    cout<<endl;
    cout<<"Negative Values Ratio: ";
    printf("%.6lf",negative);
    cout<<endl;
    cout<<"Zero Values Ratio: ";
    printf("%.6lf",zero);


    return 0;
}
