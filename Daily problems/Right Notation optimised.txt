#include <iostream>

using namespace std;

#include<vector>

vector<int> helper(int *arr,int n,int k)
{
    vector<int> v;
    if(k>n)
    {
        k=k%n;
    }
    for(int i=n-k;i<n;i++)
    {
        v.push_back(arr[i]);
    }
    for(int i=0;i<n-k;i++)
    {
        v.push_back(arr[i]);
    }
    return v;
}
int main() {
	// your code goes here
	
    
    int n;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++){
        cin>>arr[n];
    }
    int k;
    cin>>k;
    vector<int> ans=helper(arr,n,k);
    
    for(int i=0;i<n;i++)
    {
        cout<<ans[i]<<" ";
    }
	return 0;
}
