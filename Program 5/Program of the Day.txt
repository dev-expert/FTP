#include<bits/stdc++.h>
using namespace std;
int main()
{int n,rotation,k,p=0;
cout<<"Enter a Size of Array: ";
cin>>n;
cout<<"Enter Rotation: ";
cin>>rotation;
int arr[n];
cout<<"Enter Array Elements: ";
for(int i=0;i<n;i++)
{
cin>>arr[i];
}
rotation=rotation%n;
k=(n-rotation);
for(int i=0;i<n;i++)
{
    if(k<n)
    {
    cout<<arr[k++]<<" ";
    }
else
{
cout<<arr[p++]<<" ";
}
}
}
