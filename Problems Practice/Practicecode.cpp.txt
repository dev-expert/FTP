
#include <iostream>
#include<bits/stdc++.h>

using namespace std;

int main()
{
    
    
    float p=0,z=0,n=0;
    int arr[]={1,2,5,-1,2,-1,0};
    int len=sizeof(arr)/sizeof(arr[0]);
    
     for (int i=0;i<7;i++)
     {
         if(arr[i]>0)
         {
             p++;
         }
         else if(arr[i]==0){
             z++;
         }
         else
         {
            n++;   
         }
     }
     
     cout<<fixed<<setprecision(4)<<(p/len);
     cout<<fixed<<setprecision(4)<<(z/len);
     cout<<fixed<<setprecision(4)<<(n/len);
     

    return 0;
}
