#include<bits/stdc++.h>
using namespace std;


void leftRotatebyOne(int arr[], int n)
{
	int temp = arr[0], i;
	for (i = 0; i < n - 1; i++)
		arr[i] = arr[i + 1];

	arr[n-1] = temp;
}


void rightRotate(int arr[], int d, int n)
{
	for (int i = 0; i < d; i++)
		leftRotatebyOne(arr, n);
}


void printArray(int arr[], int n)
{
	for (int i = 0; i < n; i++)
		cout << arr[i] << " ";
}


int main()
{
    int n;
    int r,d;
    cin>>n;
     cin>>r;
    int arr[n];

    for (int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    
   
    d=n-r;
	
	rightRotate(arr,d, n);
	printArray(arr, n);

	return 0;
}
