#include <iostream>

using namespace std;

int main()
{
   
    int i,j;
    int n;
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++){
        cin>>arr[i];
    }
    int k;
    cin>>k;
    while(k--){
        int temp=arr[n-1];
        for(int i=n-1;i>0;i--){
            arr[i]=arr[i-1];
        }
        arr[0]=temp;
    }
    
    for(i=0;i<n;i++){
        cout<<arr[i];
    }
    return 0;
}
